<script>
const TIME_PER_QUESTION = 20;
const MAX_ATTEMPTS = 2;

const questions = [
  {
    q: "De acordo com a síntese, a TQM define-se essencialmente como:",
    opts: [
      "Um conjunto de técnicas parcelares aplicadas pelo departamento de qualidade",
      "Uma filosofia abrangente de gestão que envolve todos os membros da organização",
      "Um sistema de inspeção do produto final para detetar erros",
      "Um modelo exclusivo para grandes empresas industriais"
    ],
    ans: 1,
    exp: "A TQM não é apenas um conjunto de técnicas parcelares. Trata-se de uma filosofia abrangente de gestão que exige o envolvimento de toda a organização."
  },
  {
    q: "Qual foi o contexto histórico que impulsionou a adoção sistémica da TQM no Japão nos anos 50?",
    opts: [
      "A necessidade de cumprir normas da União Europeia de qualidade",
      "A recuperação económica e competitividade industrial após a Segunda Guerra Mundial",
      "A globalização dos mercados financeiros na década de 1950",
      "A pressão dos sindicatos japoneses por melhores condições laborais"
    ],
    ans: 1,
    exp: "O Japão adotou a TQM como filosofia cultural e estratégia de negócio para reforçar a recuperação económica e a competitividade após a Segunda Guerra Mundial."
  },
  {
    q: "O conceito japonês de Kaizen, referido na síntese, caracteriza-se por:",
    opts: [
      "Mudanças radicais e disruptivas lideradas exclusivamente pela gestão de topo",
      "Inspeção sistemática do produto final antes da entrega ao cliente",
      "Mudanças incrementais realizadas por todos os colaboradores, instalando uma cultura de melhoria constante",
      "Certificação periódica de processos segundo a norma ISO 9001"
    ],
    ans: 2,
    exp: "O Kaizen baseia-se em pequenas melhorias contínuas realizadas por todos os colaboradores, promovendo uma cultura constante de aperfeiçoamento."
  },
  {
    q: "A crítica à TQM como 'moda de gestão', sobretudo nos anos 80 e 90, é abordada na síntese como:",
    opts: [
      "Uma crítica válida que invalida definitivamente o paradigma da qualidade total",
      "Um alerta legítimo sobre aplicações superficiais, mas que não invalida o paradigma se implementado com seriedade",
      "Uma perspetiva ultrapassada que não tem qualquer fundamento nos materiais",
      "Uma consequência direta da adoção do modelo EFQM pelas empresas ocidentais"
    ],
    ans: 1,
    exp: "A crítica é vista como um alerta contra aplicações superficiais. No entanto, não invalida a TQM quando esta é aplicada com seriedade, continuidade e integração."
  },
  {
    q: "Segundo a síntese, de que forma a TQM evoluiu desde os anos 2000?",
    opts: [
      "Foi substituída integralmente pelo modelo Six Sigma, tornando-se obsoleta",
      "Manteve-se inalterada, resistindo a qualquer integração com novas metodologias",
      "Integrou-se com Lean, Six Sigma e Agile, persistindo sob designações como Excelência Operacional e modelos como o EFQM e ISO 9001",
      "Passou a aplicar-se exclusivamente ao setor dos serviços financeiros"
    ],
    ans: 2,
    exp: "Desde os anos 2000, a TQM integrou-se com metodologias como Lean, Six Sigma e Agile, mantendo relevância sob novas designações e modelos."
  }
];

let current = 0;
let score = 0;
let answered = false;
let results = [];
let attemptsLeft = MAX_ATTEMPTS;
let timer = TIME_PER_QUESTION;
let interval = null;

function updateTopBar() {
  document.getElementById("time-display").textContent = timer;
  document.getElementById("score-display").textContent = score;
}

function startTimer() {
  clearInterval(interval);
  timer = TIME_PER_QUESTION;
  updateTopBar();

  interval = setInterval(() => {
    timer--;
    updateTopBar();

    if (timer <= 0) {
      clearInterval(interval);
      if (!answered) {
        autoFailQuestion();
      }
    }
  }, 1000);
}

function autoFailQuestion() {
  answered = true;
  const q = questions[current];
  q.opts.forEach((_, idx) => {
    const b = document.getElementById("opt" + idx);
    b.disabled = true;
    if (idx === q.ans) b.classList.add("reveal");
  });

  const fb = document.getElementById("fb");
  fb.className = "feedback nok";
  fb.innerHTML = "<strong>Tempo esgotado.</strong> " + q.exp;

  results.push({ correct: false });
  document.getElementById("nxt").style.display = "inline-block";
}

function render() {
  const card = document.getElementById("quiz-card");

  if (current >= questions.length) {
    clearInterval(interval);
    document.getElementById("prog-label").textContent = "Quiz concluído";
    document.getElementById("pbar").style.width = "100%";

    const msg =
      score === 5 ? "Perfeito! Dominaste o tema." :
      score >= 4 ? "Muito bom! Quase perfeito." :
      score >= 3 ? "Bom resultado. Revê os erros." :
      "Vale a pena reler a síntese.";

    const rev = results.map((r, i) =>
      `<div class="review-item ${r.correct ? "ok" : "nok"}">
        <div class="review-q">${i + 1}. ${questions[i].q}</div>
        <div>${r.correct ? "Correto" : "Resposta correta: " + String.fromCharCode(65 + questions[i].ans) + ") " + questions[i].opts[questions[i].ans]}</div>
      </div>`
    ).join("");

    card.innerHTML = `
      <div class="score-card">
        <div><span class="score-num">${score}</span><span class="score-denom"> / ${questions.length}</span></div>
        <div class="score-msg">${msg}</div>
        <button class="btn-restart" onclick="restart()">Recomeçar</button>
        <div class="review-section">
          <div class="review-title">Resumo das respostas</div>
          ${rev}
        </div>
      </div>`;
    return;
  }

  const q = questions[current];
  document.getElementById("prog-label").textContent = `Pergunta ${current + 1} de ${questions.length}`;
  document.getElementById("pbar").style.width = ((current / questions.length) * 100) + "%";
  answered = false;
  attemptsLeft = MAX_ATTEMPTS;

  card.innerHTML = `
    <span class="tag">TQM</span>
    <div class="question-num">Pergunta ${current + 1} de ${questions.length}</div>
    <div class="question-text">${q.q}</div>
    ${q.opts.map((o, i) => `<button class="option" id="opt${i}" onclick="pick(${i})">${String.fromCharCode(65 + i)}) ${o}</button>`).join("")}
    <div class="feedback" id="fb"></div>
    <button class="btn-next" id="nxt" onclick="next()">${current === questions.length - 1 ? "Ver resultado →" : "Próxima pergunta →"}</button>
  `;

  startTimer();
}

function pick(i) {
  if (answered) return;

  const q = questions[current];
  const fb = document.getElementById("fb");

  if (i === q.ans) {
    answered = true;
    clearInterval(interval);

    q.opts.forEach((_, idx) => {
      const b = document.getElementById("opt" + idx);
      b.disabled = true;
      if (idx === q.ans) b.classList.add("correct");
    });

    fb.className = "feedback ok";
    fb.innerHTML = "<strong>Correto!</strong> " + q.exp;
    score++;
    updateTopBar();
    results.push({ correct: true });
    document.getElementById("nxt").style.display = "inline-block";
    return;
  }

  attemptsLeft--;
  document.getElementById("opt" + i).disabled = true;
  document.getElementById("opt" + i).classList.add("wrong");

  if (attemptsLeft > 0) {
    fb.className = "feedback nok";
    fb.innerHTML = `<strong>Incorreto.</strong> Ainda tens mais ${attemptsLeft} tentativa.`;
  } else {
    answered = true;
    clearInterval(interval);

    q.opts.forEach((_, idx) => {
      const b = document.getElementById("opt" + idx);
      b.disabled = true;
      if (idx === q.ans) b.classList.add("reveal");
    });

    fb.className = "feedback nok";
    fb.innerHTML = "<strong>Incorreto.</strong> " + q.exp;
    results.push({ correct: false });
    document.getElementById("nxt").style.display = "inline-block";
  }
}

function next() {
  current++;
  render();
}

function restart() {
  clearInterval(interval);
  current = 0;
  score = 0;
  answered = false;
  results = [];
  attemptsLeft = MAX_ATTEMPTS;
  render();
}

render();
</script>
